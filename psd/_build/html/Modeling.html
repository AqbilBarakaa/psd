import pandas as pd

Load Dataset (E-Coli)

# Baca dataset
df = pd.read_csv("ecoli_dataset.csv")

# Misal kolom terakhir adalah target

y = df.iloc[:, -1].values

Preprocessing

# Label encoding target
le = LabelEncoder()
y_enc = le.fit_transform(y)

# Normalisasi fitur
